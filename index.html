Solución 3: Invertir una cadena


      const invertirCadena = (cadena) => {
        return cadena.split("").reduce((acc, char) => char + acc);
      };
      
      console.log("/////////////   SOLUCION 3   /////////////");
      console.log(invertirCadena("hola")); // Resultado aloh
      console.log(invertirCadena("calabaza")); // Resultado azabalac
      console.log(invertirCadena("caballo")); // Resultado ollabac
      
      /* LOGICA: para este ejercicio se me ocurrio que un reduce serviria para iterar un array
      de caracteres e ir acumulando el valor en sentido inverso. Sin embargo, para poder 
      realizar esto debí transformar la cadena de texto en un array mediante .split("") */

Solución 4: Encontrar el palíndromo


      const esPalindromo = (cadena) => {
